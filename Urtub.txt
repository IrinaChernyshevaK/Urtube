    users = []
    videos = []

    def __init__(self, current_user=None):
        self.current_user=current_user


    def __contains__(self, item):# зачем она? если и без нее все в 1 строку решилось??
        return item in self # я верно эту тсроку записала?

    def log_in(self, nickname:str, password:str):
        for i in self.users:
            if nickname ==i.nickname:
                if hash(password) == i.password:
                    self.current_user=User(nickname,password, i.age)
                    break
                else:
                    print('неверный пароль')
                    self.log_in(nickname, password)
            else:
                continue
            print('Пройдите регистрацию')


    def register(self, nickname:str, password:str, age:int|float):
        list=[]
        for i in self.users:
            if nickname == i.nickname:
                list.append(True)
                print(f'Пользователь {nickname} уже существует')
                break
        if True not in list:
            self.users.append(User(nickname, password, age))
            self.log_in(nickname, password)


    def log_out(self):
        self.current_user=None

    def add(self, *args):
        for i in args:
            if i.title not in self.videos:
                    self.videos.append(i)


    def get_videos(self, word:str):
        res = []
        for i in self.videos:
            if word.lower() in i.title.lower():
                res.append(i.title)
        return res

    def __eq__(self,other): #аналогично __contens__
        return self == other

    def watch_video(self, film:str):
        if self.current_user:
            film_bool=[]
            for i in self.videos:
                if film == i.title:
                    film_bool.append(True)
                    if i.adult_mode == True and self.current_user.age<18:
                        print('Вам нет 18 лет, пожалуйста покиньте страницу')
                    else:
                        for j in range(i.duration+1):
                            print(i.time_now, end=' ')
                            i.time_now+=1
                            time.sleep(1)
                        print('конец видео')
                        i.time_now=0
                        break
                else:
                    continue
            if True not in film_bool:
                    print(f'Фильма {film} нет')
        else:
            print('Войдите в аккаунт, чтобы смотреть видео')
